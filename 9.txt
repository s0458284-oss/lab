<script>
$(document).ready(function(){

    $("#appendButton").click(function(){
        $("#paragraph").append(" Appended text.");
        $("#list").append("<li>New appended list item</li>");
    });

    $("#animateButton").click(function(){
        $("#box").css({
            width:"100px",
            height:"100px",
            opacity:1,
            background:"blue"
        }).animate({
            width:"200px",
            height:"200px",
            opacity:0.5
        },1000,function(){
            $(this).css("background","green");
        });
    });

});
